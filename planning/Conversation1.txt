serve(async (req) => {

      completed_at: new Date().toISOString()

      headers: { 'Content-Type': 'application/json' }
    })
  }

  await triggerAIAnalysis(response.id)
  
  return new Response(JSON.stringify({ success: true, response_id: response.id }), {
    headers: { 'Content-Type': 'application/json' }
  })

typescript// lambda/ai-analysis/sentiment-analyzer.ts
import { SQSHandler } from 'aws-lambda'
import { createClient } from '@supabase/supabase-js'
import { BedrockRuntimeClient, InvokeModelCommand } from '@aws-sdk/client-bedrock-runtime'

const supabase = createClient(process.env.SUPABASE_URL!, process.env.SUPABASE_SERVICE_KEY!)
const bedrock = new BedrockRuntimeClient({ region: 'us-east-1' })

export const handler: SQSHandler = async (event) => {
  for (const record of event.Records) {
    const { response_id } = JSON.parse(record.body)
    
    // Get response from database
    const { data: response } = await supabase
      .from('responses')
      .select('*, survey:surveys(*)')
      .eq('id', response_id)
      .single()
    
    if (!response) continue
    
    // Extract open-ended text from answers
    const openEndedText = extractOpenEndedAnswers(response.answers, response.survey.questions)
    
    if (!openEndedText) continue
    
    // Call Claude via Bedrock for analysis
    const prompt = `Analyze the following customer feedback and provide:
1. Overall sentiment score (-1.0 to 1.0)
2. Sentiment label (very_negative, negative, neutral, positive, very_positive)
3. Detected emotions (array)
4. Main themes (array)
5. Urgency score (0-10)
6. Whether this requires urgent attention (boolean)

Feedback: "${openEndedText}"

Respond in JSON format only.`

    const command = new InvokeModelCommand({
      modelId: 'anthropic.claude-3-5-sonnet-20241022-v2:0',
      body: JSON.stringify({
        anthropic_version: 'bedrock-2023-05-31',
        max_tokens: 1000,
        messages: [{ role: 'user', content: prompt }]
      })
    })
    
    const bedrockResponse = await bedrock.send(command)
    const aiResult = JSON.parse(new TextDecoder().decode(bedrockResponse.body))
    const analysis = JSON.parse(aiResult.content[0].text)
    
    // Update response with AI analysis
    await supabase
      .from('responses')
      .update({
        sentiment_score: analysis.sentiment_score,
        sentiment_label: analysis.sentiment_label,
        emotions: analysis.emotions,
        urgency_score: analysis.urgency_score,
        is_urgent: analysis.is_urgent,
        updated_at: new Date().toISOString()
      })
      .eq('id', response_id)
    
    // Extract and save themes
    await processThemes(response_id, response.survey_id, analysis.themes)
    
    // Send alerts if urgent
    if (analysis.is_urgent) {
      await sendUrgentAlert(response)
    }
  }
}

typescript// cdk/lib/lambda-stack.ts
import * as cdk from 'aws-cdk-lib'
import * as lambda from 'aws-cdk-lib/aws-lambda'
import * as sqs from 'aws-cdk-lib/aws-sqs'

export class AIAnalysisStack extends cdk.Stack {
  constructor(scope: Construct, id: string, props?: cdk.StackProps) {
    super(scope, id, props)
    
    // SQS Queue for async processing
    const analysisQueue = new sqs.Queue(this, 'AnalysisQueue', {
      visibilityTimeout: cdk.Duration.seconds(300),
      retentionPeriod: cdk.Duration.days(4)
    })
    
    // Lambda function with cost optimization
    const sentimentAnalyzer = new lambda.Function(this, 'SentimentAnalyzer', {
      runtime: lambda.Runtime.NODEJS_20_X,
      code: lambda.Code.fromAsset('lambda/ai-analysis'),
      handler: 'sentiment-analyzer.handler',
      timeout: cdk.Duration.seconds(60),
      memorySize: 512, // Optimize based on actual needs
      
      // Reserve concurrency to control costs
      reservedConcurrentExecutions: 10,
      
      environment: {
        SUPABASE_URL: process.env.SUPABASE_URL!,
        SUPABASE_SERVICE_KEY: process.env.SUPABASE_SERVICE_KEY!
      },
      
      events: [
        new lambdaEventSources.SqsEventSource(analysisQueue, {
          batchSize: 10 // Process multiple responses at once
        })
      ]
    })
    
    // Grant Bedrock access
    sentimentAnalyzer.addToRolePolicy(new iam.PolicyStatement({
      actions: ['bedrock:InvokeModel'],
      resources: ['*']
    }))
  }
}

### **6. Cost Optimization Strategies**

#### **Database (Supabase)**
- **Free tier**: Up to 500MB database, 1GB file storage, 2GB bandwidth
- **Pro tier** ($25/mo): 8GB database, 100GB file storage, 50GB bandwidth
- **Optimization**:
  - Use JSONB for flexible schema (avoid excessive tables)
  - Implement proper indexes
  - Archive old responses to S3 (cheap storage)
  - Use Supabase's built-in connection pooling

#### **AI/ML (AWS Bedrock)**
- **Claude Sonnet pricing**: ~$3 per 1M input tokens, ~$15 per 1M output tokens
- **Optimization**:
  - Batch process responses (analyze 10-20 at once)
  - Use smaller prompts (be concise)
  - Cache common analysis results
  - Use AWS Comprehend for simple sentiment ($0.0001 per unit) and Bedrock for complex analysis
  - Implement request throttling to control costs

#### **Lambda**
- **Free tier**: 1M requests/month, 400,000 GB-seconds
- **Optimization**:
  - Use ARM architecture (20% cheaper)
  - Right-size memory (512MB is often sufficient)
  - Reserve concurrency to prevent runaway costs
  - Use SQS for batching (process 10 responses per invocation)

#### **Email/SMS (SES/SNS)**
- **SES**: $0.10 per 1,000 emails
- **SNS**: $0.00645 per SMS (US)
- **Optimization**:
  - Use SES for transactional emails (much cheaper than SendGrid)
  - Implement smart reminder logic (don't spam)
  - Use email for most communications, SMS only for critical

#### **Storage (S3)**
- **Standard**: $0.023 per GB/month
- **Optimization**:
  - Use S3 Intelligent-Tiering for exports
  - Set lifecycle policies (delete old exports after 30 days)
  - Use CloudFront CDN (cheaper egress)
